point = [9; 1];
H = [1, 0; 0, 9];
figure
ezcontour('x^2/2+9*y^2/2',[-9, 9, -6, 6])

% steptest decent method
sdm_points = [9; 1];
count = 0;
while(norm(point) > 1e-5)
    count = count + 1;
    g = [point(1); 9*point(2)];
    point = point - g'*g/(g'*H*g).*g;
    sdm_points = [sdm_points, point];
end
hold on
plot(sdm_points(1, :), sdm_points(2, :),'-','LineWidth',3);
count
\end{lstlisting}
